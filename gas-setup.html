// ============================================================
// 청솔행정사사무소 Google Sheets 백엔드 API 시스템
// ============================================================

const NOTIFY_EMAIL = 'cirrus01@naver.com';

function doOptions(e) {
  return ContentService.createTextOutput("").setMimeType(ContentService.MimeType.TEXT);
}

// 1. GET 요청 (모든 조회, 등록, 수정 처리 - CORS 완벽 우회)
function doGet(e) {
  try {
    const action = e.parameter.action || "read";
    const tableName = e.parameter.table || "Applications";
    
    if (action === "create" || action === "update") {
      const dataStr = e.parameter.data;
      if (!dataStr) return jsonResponse({ status: 'error', message: '데이터가 없습니다.' });
      
      const data = JSON.parse(dataStr);
      const sheet = initSheet(tableName, data);
      
      if (action === "update") {
        updateApplication(sheet, e.parameter.id || data.id, data.payload || data);
        return jsonResponse({ status: 'ok', message: 'Updated' });
      } else {
        createApplication(sheet, data);
        try {
          if (tableName === "Applications") sendNotificationEmail(data);
        } catch(emailErr) {}
        return jsonResponse({ status: 'ok', message: 'Created' });
      }
    }
    
    // read 동작
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(tableName);
    if (!sheet) return jsonResponse({ data: [] });
    
    const data = sheet.getDataRange().getValues();
    if (data.length <= 1) return jsonResponse({ data: [] });
    
    const headers = data[0];
    const rows = data.slice(1);
    const result = rows.map(row => {
      let obj = {};
      headers.forEach((h, i) => { if(h) obj[h] = row[i]; });
      return obj;
    });
    
    return jsonResponse({ data: result });
  } catch (err) {
    return jsonResponse({ status: 'error', message: err.message });
  }
}

// 2. POST 요청 (하위 호환성 유지)
function doPost(e) {
  try {
    const rawData = (e.parameter && e.parameter.data) ? e.parameter.data : (e.postData ? e.postData.contents : "{}");
    const data = JSON.parse(rawData);
    
    const tableName = data.table || "Applications";
    const sheet = initSheet(tableName, data);
    
    if (data.action === "update") {
      updateApplication(sheet, data.id, data.payload || data);
      return jsonResponse({ status: 'ok', message: 'Updated' });
    } else {
      createApplication(sheet, data);
      try {
        if (tableName === "Applications") sendNotificationEmail(data);
      } catch (emailErr) {}
      return jsonResponse({ status: 'ok', message: 'Created' });
    }
  } catch (err) {
    return jsonResponse({ status: 'error', message: err.message });
  }
}

function initSheet(sheetName, data) {
  let ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(sheetName);
  
  const defaultHeaders = [
    "id", "app_number", "applicant_type", "company_name", "manager_name", 
    "phone", "email", "business_type", "business_reg_no", "category", 
    "sub_service", "content", "current_status", "deadline", "reply_method", 
    "is_urgent", "status", "admin_memo", "created_at"
  ];

  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    sheet.appendRow(sheetName === "Applications" ? defaultHeaders : ["id", "created_at"]);
  } else {
    const firstRow = sheet.getRange(1, 1, 1, Math.max(1, sheet.getLastColumn())).getValues()[0];
    if (!firstRow || !firstRow.join("").trim()) {
      sheet.getRange(1, 1, 1, defaultHeaders.length).setValues([defaultHeaders]);
    }
  }
  return sheet;
}

function createApplication(sheet, d) {
  const lastCol = sheet.getLastColumn();
  if (lastCol === 0) return;
  const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
  
  d.id = 'ID_' + new Date().getTime();
  d.created_at = d.submitted_at || new Date().toISOString();
  
  const rowData = headers.map(h => d[h] === undefined ? '' : String(d[h]));
  sheet.appendRow(rowData);
}

function updateApplication(sheet, id, payload) {
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const idIndex = headers.indexOf("id");
  if (idIndex === -1) return;
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][idIndex] === id) {
      headers.forEach((h, colIndex) => {
        if (payload[h] !== undefined) {
          sheet.getRange(i + 1, colIndex + 1).setValue(payload[h]);
        }
      });
      break;
    }
  }
}

function sendNotificationEmail(d) {
  const isUrgent = (d.is_urgent === true || d.is_urgent === 'true');
  const subject = (isUrgent ? '🚨 [긴급] ' : '') + '[청솔행정사] 새 민원 접수 - ' + (d.app_number);
  const body = '안녕하세요.\n\n새 민원 접수가 구글 스프레드시트에 등록되었습니다.\n접수번호: ' + d.app_number + '\n회사명: ' + d.company_name + '\n연락처: ' + d.phone;
  GmailApp.sendEmail(NOTIFY_EMAIL, subject, body);
}

function jsonResponse(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}

// 최초 1회 실행하여 모든 권한(스프레드시트, Gmail) 승인받기
function setupAndTest() {
  initSheet("Applications");
  try {
    GmailApp.sendEmail(Session.getActiveUser().getEmail(), "청솔행정사 시스템 연동 완료", "구글 시트 및 이메일 알림 권한이 성공적으로 승인되었습니다.");
  } catch(e) {}
}

4
최초 1회 함수 실행 (매우 중요 ⭐️)

코드 상단의 저장(디스켓 아이콘)을 누르고, 메뉴에서 setupAndTest 함수를 선택한 뒤 ▶ 실행 버튼을 누릅니다.

Google 권한 팝업이 뜨면 권한 검토 ➔ 고급 ➔ 해당 스크립트로 이동(안전하지 않음) ➔ 허용 순으로 눌러줍니다. 이 작업을 완료해야 에러가 나지 않습니다.

웹 앱(Web App)으로 배포

우측 상단 파란색 배포 버튼 ➔ 새 배포를 클릭합니다.

'유형 선택'에서 톱니바퀴를 눌러 웹 앱을 선택하고 다음과 같이 설정한 후 배포를 클릭합니다.

  • 실행할 사용자: 나(본인 계정)
  • 액세스 권한: 모든 사용자 (반드시 '모든 사용자'여야 합니다)
